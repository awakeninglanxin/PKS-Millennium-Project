BASE = 7.83  # 地球舒曼基频 (MHz/10^6 = 7.83e-6 MHz)

def min_distance_mhz(freq_mhz, base, scale='2^n'):
    """找到频率与 base × scale 的最小偏差"""
    if scale == '2^n':
        n_closest = round(math.log2(freq_mhz / BASE))
        nearest = BASE * (2 ** n_closest)
    else:  # φ^n
        n_closest = round(math.log(freq_mhz / BASE, PHI))
        nearest = BASE * (PHI ** n_closest)
    return freq_mhz - nearest, nearest, n_closest
